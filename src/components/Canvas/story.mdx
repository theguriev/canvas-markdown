import { useState, useMemo } from 'react'
import { Meta, Story, Canvas, Props } from '@storybook/addon-docs/blocks'
import { parserText } from '@/markdown/text'
import { render } from '@/markdown/render'
import { argTypes, args } from './argTypes'
import { DEVICE_PIXEL_RATIO } from '@/constants'
import { useZoom, usePosition, usePan } from '@/hooks'
import CanvasMD from '.'

export const StoryBookWrapper = ({
    text,
    autoSize,
    blockRadius,
    codespanStrokeStyle,
    codespanFillStyle,
    linkFillStyle,
    hrStrokeStyle,
    blockquoteWidth,
    blockquoteFillStyle,
    blockquoteStrokeStyle,
    codeFillStyle,
    codeStrokeStyle,
    listRadius,
    listFillStyle,
    listMarginLeft,
    listMarginBottom,
    fontFamily,
    fontPadding,
    fontSize,
    fontLineHeight,
    fontHeadingSize1,
    fontHeadingSize2,
    fontHeadingSize3,
    fontHeadingSize4,
    fontHeadingSize5,
    fontHeadingSize6
}) => {
    const [dom, setDom] = useState(null)
    const parts = parserText({
        text,
        width: 600,
        theme: {
            blockRadius,
            codespanStrokeStyle,
            codespanFillStyle,
            linkFillStyle,
            hrStrokeStyle,
            blockquoteWidth,
            blockquoteFillStyle,
            blockquoteStrokeStyle,
            codeFillStyle,
            codeStrokeStyle,
            listRadius,
            listFillStyle,
            listMarginLeft,
            listMarginBottom,
            fontFamily,
            fontPadding,
            fontSize,
            fontLineHeight,
            fontHeadingSize1,
            fontHeadingSize2,
            fontHeadingSize3,
            fontHeadingSize4,
            fontHeadingSize5,
            fontHeadingSize6
        }
    })
    const {
        scale,
        offsetX,
        offsetY
    } = useZoom({ startScale: 1, dom })
    const {
        x,
        y,
        hoverX,
        hoverY
    } = usePosition({ dom, offsetX, offsetY })
    const { isSpacePressing, isMouseDown } = usePan({ dom, offsetX, offsetY })
    const cursor = useMemo(
        () => {
            if (isSpacePressing && isMouseDown) {
                return 'grabbing'
            }
            if (isSpacePressing) {
                return 'grab'
            }
            return 'default'
        },
        [isSpacePressing, isMouseDown]
    )
    const draw = ({ context, dom: cnvs }) => {
        context.setTransform(1, 0, 0, 1, 0, 0)
        context.clearRect(
            0,
            0,
            cnvs.width,
            cnvs.height
        )
        context.setTransform(
            DEVICE_PIXEL_RATIO,
            0,
            0,
            DEVICE_PIXEL_RATIO,
            offsetX.current,
            offsetY.current
        )
        context.scale(scale.current, scale.current)
        render({
            ctx: context,
            parts,
            cursorX: { current: 0 },
            cursorY: { current: 0 },
            hoveredNode: { current: 0 },
            hoveredPath: { current: 0 },
            hoveredPlusArea: { current: 0 },
            isCtrlPressing: { current: false }
        })
    }
    return (<CanvasMD draw={draw} ref={setDom} cursor={cursor} />)
}

<Meta
    title="Canvas"
    component={StoryBookWrapper}
    argTypes={argTypes}
/>

<Canvas>
    <Story name="Default" args={args}>
        {(props) => {
            const draw = ({ context }) => {
                context.beginPath()
                context.arc(75, 75, 50, 0, 2 * Math.PI)
                context.stroke()
            }
            return (<CanvasMD draw={draw} />)
        }}
    </Story>
</Canvas>
